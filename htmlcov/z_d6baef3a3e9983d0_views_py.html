61class SupersetView(GenericAPIView): 

62 """ 

63 Superset-related endpoints provided by the aspects platform plugin. 

64 """ 

65 

66 authentication_classes = (SessionAuthentication,) 

67 permission_classes = ( 

68 permissions.IsAuthenticated, 

69 IsStaffOrReadOnly | IsCourseStaffInstructor, 

70 ) 

71 

72 lookup_field = "course_id" 

73 

74 def get_object(self): 

75 """ 

76 Return a Course-like object for the requested course_id. 

77 """ 

78 course_id = self.kwargs.get(self.lookup_field, "") 

79 try: 

80 course_key = CourseKey.from_string(course_id) 

81 except InvalidKeyError as exc: 

82 raise NotFound( 

83 _("Invalid course id: '{course_id}'").format(course_id=course_id) 

84 ) from exc 

85 

86 # Fetch the CourseOverview (if we're running in edx-platform) 

87 display_name = "" 

88 CourseOverview = get_model("course_overviews") 

89 if CourseOverview: 

90 try: 

91 course_overview = CourseOverview.objects.get(id=course_key) 

92 display_name = course_overview.display_name 

93 except CourseOverview.DoesNotExist as exc: 

94 raise NotFound( 

95 _("Course not found: '{course_id}'").format(course_id=course_id) 

96 ) from exc 

97 

98 course = Course(course_id=course_key, display_name=display_name) 

99 

100 # May raise a permission denied 

101 self.check_object_permissions(self.request, course) 

102 

103 return course_key 

104 

105 def get(self, request, *args, **kwargs): 

106 """ 

107 Return a guest token for accessing the Superset API. 

108 """ 

109 course = self.get_object() 

110 

111 dashboards = settings.ASPECTS_INSTRUCTOR_DASHBOARDS 

112 extra_filters_format = settings.SUPERSET_EXTRA_FILTERS_FORMAT 

113 

114 try: 

115 guest_token = generate_guest_token( 

116 user=request.user, 

117 course=course, 

118 dashboards=dashboards, 

119 filters=DEFAULT_FILTERS_FORMAT + extra_filters_format, 

120 ) 

121 except ImproperlyConfigured as exc: 

122 raise APIException() from exc 

123 

124 return Response({"guestToken": guest_token})
